163 template <typename Type, typename PassType, typename ReturnType>
164 void MAnimation<Type, PassType, ReturnType>::update(
165  const MTime& aTime)
166  {
167  unsigned int count = theKeys.getCount();
168 
169  if (count <= 1)
170  {
171  if (count == 1)
172  theValue = theKeys[0].theValue;
173  return;
174  }
175 
176  MTime curTime = aTime;
177  if (theMode == PLAY_MODE_REPEAT)
178  {
179  MTime range;
180  range = (theKeys[count - 1].theTime - theKeys[0].theTime);
181 
182  double currSecs = curTime.getSeconds();
183  currSecs = currSecs - range.getSeconds() * floor((currSecs - theKeys[0].theTime.getSeconds()) / (range.getSeconds()));
184  curTime = MTime(currSecs);
185  }
186 
187  // Find left and right keys in neighbourhood of aTime parameter
188  MDRKey* left = NULL;
189  MDRKey* right = NULL;
190  for (unsigned int index = 1; index < count; index++)
191  {
192  if (curTime >= theKeys[index - 1].theTime &&
193  curTime <= theKeys[index].theTime)
194  {
195  left = &theKeys[index - 1];
196  right = &theKeys[index];
197  break;
198  }
199  }
200 
201  // If neighbourhood was not found then do not calculate animation
202  if (left == NULL ||
203  right == NULL)
204  return;
205 
206  double coefficient = (curTime.getSeconds() - left->theTime.getSeconds()) / (right->theTime.getSeconds() - left->theTime.getSeconds());
207  lerp(coefficient, left->theValue, right->theValue, theValue);
208  }
